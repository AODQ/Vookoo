252  Window(const vk::Instance &instance, const vk::Device &device, const vk::PhysicalDevice &physicalDevice, uint32_t graphicsQueueFamilyIndex, GLFWwindow *window) {
253  device_ = device;
254 
255 #ifdef VK_USE_PLATFORM_WIN32_KHR
256  auto module = GetModuleHandle(nullptr);
257  auto handle = glfwGetWin32Window(window);
258  auto ci = vk::Win32SurfaceCreateInfoKHR{{}, module, handle};
259  surface_ = instance.createWin32SurfaceKHRUnique(ci);
260 #endif
261 #ifdef VK_USE_PLATFORM_XLIB_KHR
262  auto display = glfwGetX11Display();
263  auto x11window = glfwGetX11Window(window);
264  auto ci = vk::XlibSurfaceCreateInfoKHR{{}, display, x11window};
265  surface_ = instance.createXlibSurfaceKHRUnique(ci);
266 #endif
267 
268  presentQueueFamily_ = 0;
269  auto &pd = physicalDevice;
270  auto qprops = pd.getQueueFamilyProperties();
271  for (uint32_t qi = 0; qi != qprops.size(); ++qi) {
272  auto &qprop = qprops[qi];
273  VkBool32 presentSupport = false;
274  if (pd.getSurfaceSupportKHR(qi, *surface_)) {
275  presentQueue_ = device.getQueue(qi, 0);
276  presentQueueFamily_ = qi;
277  }
278  }
279 
280  if (!presentQueue_) {
281  std::cout << "No Vulkan queues found\n";
282  return;
283  }
284 
285  auto fmts = pd.getSurfaceFormatsKHR(*surface_);
286  swapchainImageFormat_ = fmts[0].format;
287  swapchainColorSpace_ = fmts[0].colorSpace;
288  if (fmts.size() == 1 && swapchainImageFormat_ == vk::Format::eUndefined) {
289  swapchainImageFormat_ = vk::Format::eB8G8R8A8Unorm;
290  swapchainColorSpace_ = vk::ColorSpaceKHR::eSrgbNonlinear;
291  } else {
292  for (auto &fmt : fmts) {
293  if (fmt.format == vk::Format::eB8G8R8A8Unorm) {
294  swapchainImageFormat_ = fmt.format;
295  swapchainColorSpace_ = fmt.colorSpace;
296  }
297  }
298  }
299 
300  auto surfaceCaps = pd.getSurfaceCapabilitiesKHR(*surface_);
301  width_ = surfaceCaps.currentExtent.width;
302  height_ = surfaceCaps.currentExtent.height;
303 
304  auto pms = pd.getSurfacePresentModesKHR(*surface_);
305  vk::PresentModeKHR presentMode = pms[0];
306  if (std::find(pms.begin(), pms.end(), vk::PresentModeKHR::eMailbox) != pms.end()) {
307  presentMode = vk::PresentModeKHR::eMailbox;
308  } else if (std::find(pms.begin(), pms.end(), vk::PresentModeKHR::eFifo) != pms.end()) {
309  presentMode = vk::PresentModeKHR::eFifo;
310  }
311  std::cout << "using " << vk::to_string(presentMode) << "\n";
312 
313  vk::SwapchainCreateInfoKHR swapinfo{};
314  std::array<uint32_t, 2> queueFamilyIndices = { graphicsQueueFamilyIndex, presentQueueFamily_ };
315  bool sameQueues = queueFamilyIndices[0] == queueFamilyIndices[1];
316  vk::SharingMode sharingMode = !sameQueues ? vk::SharingMode::eConcurrent : vk::SharingMode::eExclusive;
317  swapinfo.imageExtent = surfaceCaps.currentExtent;
318  swapinfo.surface = *surface_;
319  swapinfo.minImageCount = surfaceCaps.minImageCount + 1;
320  swapinfo.imageFormat = swapchainImageFormat_;
321  swapinfo.imageColorSpace = swapchainColorSpace_;
322  swapinfo.imageExtent = surfaceCaps.currentExtent;
323  swapinfo.imageArrayLayers = 1;
324  swapinfo.imageUsage = vk::ImageUsageFlagBits::eColorAttachment;
325  swapinfo.imageSharingMode = sharingMode;
326  swapinfo.queueFamilyIndexCount = !sameQueues ? 2 : 0;
327  swapinfo.pQueueFamilyIndices = queueFamilyIndices.data();
328  swapinfo.preTransform = surfaceCaps.currentTransform;;
329  swapinfo.compositeAlpha = vk::CompositeAlphaFlagBitsKHR::eOpaque;
330  swapinfo.presentMode = presentMode;
331  swapinfo.clipped = 1;
332  swapinfo.oldSwapchain = vk::SwapchainKHR{};
333  swapchain_ = device.createSwapchainKHRUnique(swapinfo);
334 
335  images_ = device.getSwapchainImagesKHR(*swapchain_);
336  for (auto &img : images_) {
337  vk::ImageViewCreateInfo ci{};
338  ci.image = img;
339  ci.viewType = vk::ImageViewType::e2D;
340  ci.format = swapchainImageFormat_;
341  ci.subresourceRange = {vk::ImageAspectFlagBits::eColor, 0, 1, 0, 1};
342  imageViews_.emplace_back(device.createImageView(ci));
343  }
344 
345  auto memprops = physicalDevice.getMemoryProperties();
346  depthStencilImage_ = vku::DepthStencilImage(device, memprops, width_, height_);
347 
348  // This subpass dependency handles the transition from ePresentSrcKHR to eUndefined
349  // at the start of rendering.
350  vk::SubpassDependency dependency = {};
351  dependency.srcSubpass = VK_SUBPASS_EXTERNAL;
352  dependency.dstSubpass = 0;
353  dependency.srcStageMask = vk::PipelineStageFlagBits::eColorAttachmentOutput;
354  dependency.srcAccessMask = {};
355  dependency.dstStageMask = vk::PipelineStageFlagBits::eColorAttachmentOutput;
356  dependency.dstAccessMask = vk::AccessFlagBits::eColorAttachmentRead|vk::AccessFlagBits::eColorAttachmentWrite;
357 
358  // At the start, the buffer is in eUndefined layout.
359  // We will clear the buffer at the start (eClear)
360  // We will write the result out at the end (eStore)
361  // After this renderpass, we will switch to ePresentSrcKHR ready for the swap.
362  vk::AttachmentDescription colourDesc{};
363  colourDesc.format = swapchainImageFormat_;
364  colourDesc.samples = vk::SampleCountFlagBits::e1;
365  colourDesc.loadOp = vk::AttachmentLoadOp::eClear;
366  colourDesc.storeOp = vk::AttachmentStoreOp::eStore;
367  colourDesc.stencilLoadOp = vk::AttachmentLoadOp::eDontCare;
368  colourDesc.stencilStoreOp = vk::AttachmentStoreOp::eDontCare;
369  colourDesc.initialLayout = vk::ImageLayout::eUndefined;
370  colourDesc.finalLayout = vk::ImageLayout::ePresentSrcKHR;
371 
372  // At the start, the buffer is in eUndefined layout.
373  // We will clear the buffer at the start (eClear)
374  vk::AttachmentDescription depthDesc{};
375  depthDesc.format = depthStencilImage_.format();
376  depthDesc.samples = vk::SampleCountFlagBits::e1;
377  depthDesc.loadOp = vk::AttachmentLoadOp::eClear;
378  depthDesc.storeOp = vk::AttachmentStoreOp::eDontCare;
379  depthDesc.stencilLoadOp = vk::AttachmentLoadOp::eDontCare;
380  depthDesc.stencilStoreOp = vk::AttachmentStoreOp::eDontCare;
381  depthDesc.initialLayout = vk::ImageLayout::eUndefined;
382  depthDesc.finalLayout = vk::ImageLayout::eDepthStencilAttachmentOptimal;
383 
384  vku::RenderpassMaker rpm;
385  rpm.attachmentDescription(colourDesc);
386  rpm.attachmentDescription(depthDesc);
387  rpm.beginSubpass(vk::PipelineBindPoint::eGraphics);
388  rpm.subpassColorAttachment(vk::ImageLayout::eColorAttachmentOptimal, 0);
389  rpm.subpassDepthStencilAttachment(vk::ImageLayout::eDepthStencilAttachmentOptimal, 1);
390  rpm.subpassDependency(dependency);
391  renderPass_ = rpm.createUnique(device);
392 
393  for (int i = 0; i != imageViews_.size(); ++i) {
394  vk::ImageView attachments[2] = {imageViews_[i], depthStencilImage_.imageView()};
395  vk::FramebufferCreateInfo fbci{{}, *renderPass_, 2, attachments, width_, height_, 1 };
396  framebuffers_.push_back(device.createFramebufferUnique(fbci));
397  }
398 
399  vk::SemaphoreCreateInfo sci;
400  imageAcquireSemaphore_ = device.createSemaphoreUnique(sci);
401  commandCompleteSemaphore_ = device.createSemaphoreUnique(sci);
402  presubmitSemaphore_ = device.createSemaphoreUnique(sci);
403 
404  typedef vk::CommandPoolCreateFlagBits ccbits;
405 
406  vk::CommandPoolCreateInfo cpci{ ccbits::eTransient|ccbits::eResetCommandBuffer, graphicsQueueFamilyIndex };
407  commandPool_ = device.createCommandPoolUnique(cpci);
408 
409  vk::CommandBufferAllocateInfo cbai{ *commandPool_, vk::CommandBufferLevel::ePrimary, (uint32_t)framebuffers_.size() };
410  drawBuffers_ = device.allocateCommandBuffersUnique(cbai);
411  presubmitBuffers_ = device.allocateCommandBuffersUnique(cbai);
412  for (int i = 0; i != drawBuffers_.size(); ++i) {
413  vk::FenceCreateInfo fci;
414  fci.flags = vk::FenceCreateFlagBits::eSignaled;
415  commandBufferFences_.emplace_back(device.createFence(fci));
416  }
417 
418  ok_ = true;
419  }

469  void draw(const vk::Device &device, const vk::Queue &graphicsQueue, const std::function<void (vk::CommandBuffer pscb, int imageIndex)> &preSubmit = defaultPreSubmitFunc) {
470  static auto start = std::chrono::high_resolution_clock::now();
471  auto time = std::chrono::high_resolution_clock::now();
472  auto delta = time - start;
473  start = time;
474  //std::cout << std::chrono::duration_cast<std::chrono::microseconds>(delta).count() << "us frame time\n";
475 
476  auto umax = std::numeric_limits<uint64_t>::max();
477  uint32_t imageIndex = 0;
478  device.acquireNextImageKHR(*swapchain_, umax, *imageAcquireSemaphore_, vk::Fence(), &imageIndex);
479 
480  vk::PipelineStageFlags waitStages = vk::PipelineStageFlagBits::eColorAttachmentOutput;
481  vk::Semaphore ccSema = *commandCompleteSemaphore_;
482  vk::Semaphore iaSema = *imageAcquireSemaphore_;
483  vk::Semaphore psSema = *presubmitSemaphore_;
484  vk::CommandBuffer cb = *drawBuffers_[imageIndex];
485  vk::CommandBuffer pscb = *presubmitBuffers_[imageIndex];
486 
487  vk::Fence cbFence = commandBufferFences_[imageIndex];
488  device.waitForFences(cbFence, 1, umax);
489  device.resetFences(cbFence);
490 
491  preSubmit(pscb, imageIndex);
492 
493  vk::SubmitInfo submit;
494  submit.waitSemaphoreCount = 1;
495  submit.pWaitSemaphores = &iaSema;
496  submit.pWaitDstStageMask = &waitStages;
497  submit.commandBufferCount = 1;
498  submit.pCommandBuffers = &pscb;
499  submit.signalSemaphoreCount = 1;
500  submit.pSignalSemaphores = &psSema;
501  graphicsQueue.submit(1, &submit, vk::Fence{});
502 
503  submit.waitSemaphoreCount = 1;
504  submit.pWaitSemaphores = &psSema;
505  submit.pWaitDstStageMask = &waitStages;
506  submit.commandBufferCount = 1;
507  submit.pCommandBuffers = &cb;
508  submit.signalSemaphoreCount = 1;
509  submit.pSignalSemaphores = &ccSema;
510  graphicsQueue.submit(1, &submit, cbFence);
511 
512  vk::PresentInfoKHR presentInfo;
513  vk::SwapchainKHR swapchain = *swapchain_;
514  presentInfo.pSwapchains = &swapchain;
515  presentInfo.swapchainCount = 1;
516  presentInfo.pImageIndices = &imageIndex;
517  presentInfo.waitSemaphoreCount = 1;
518  presentInfo.pWaitSemaphores = &ccSema;
519  presentQueue_.presentKHR(presentInfo);
520  }
