46 inline int findMemoryTypeIndex(const vk::PhysicalDeviceMemoryProperties &memprops, uint32_t memoryTypeBits, vk::MemoryPropertyFlags search) {
47  for (int i = 0; i != memprops.memoryTypeCount; ++i, memoryTypeBits >>= 1) {
48  if (memoryTypeBits & 1) {
49  if ((memprops.memoryTypes[i].propertyFlags & search) == search) {
50  return i;
51  }
52  }
53  }
54  return -1;
55 }

183  // A variable in a shader.
184  struct Variable {
185  // The name of the variable from the GLSL/HLSL
186  std::string debugName;
187 
188  // The internal name (integer) of the variable
189  int name;
190 
191  // The location in the binding.
192  int location;
193 
194  // The binding in the descriptor set or I/O channel.
195  int binding;
196 
197  // The descriptor set (for uniforms)
198  int set;
199  int instruction;
200 
201  // Storage class of the variable, eg. spv::StorageClass::Uniform
202  spv::StorageClass storageClass;
203  };
204 
205  // Get a list of variables from the shader.
206  //
207  // This exposes the Uniforms, inputs, outputs, push constants.
208  // See spv::StorageClass for more details.
209  std::vector<Variable> getVariables() const {
210  auto bound = s.opcodes_[3];
211 
212  std::unordered_map<int, int> bindings;
213  std::unordered_map<int, int> locations;
214  std::unordered_map<int, int> sets;
215  std::unordered_map<int, std::string> debugNames;
216 
217  for (int i = 5; i != s.opcodes_.size(); i += s.opcodes_[i] >> 16) {
218  spv::Op op = spv::Op(s.opcodes_[i] & 0xffff);
219  if (op == spv::Op::OpDecorate) {
220  int name = s.opcodes_[i + 1];
221  auto decoration = spv::Decoration(s.opcodes_[i + 2]);
222  if (decoration == spv::Decoration::Binding) {
223  bindings[name] = s.opcodes_[i + 3];
224  } else if (decoration == spv::Decoration::Location) {
225  locations[name] = s.opcodes_[i + 3];
226  } else if (decoration == spv::Decoration::DescriptorSet) {
227  sets[name] = s.opcodes_[i + 3];
228  }
229  } else if (op == spv::Op::OpName) {
230  int name = s.opcodes_[i + 1];
231  debugNames[name] = (const char *)&s.opcodes_[i + 2];
232  }
233  }
234 
235  std::vector<Variable> result;
236  for (int i = 5; i != s.opcodes_.size(); i += s.opcodes_[i] >> 16) {
237  spv::Op op = spv::Op(s.opcodes_[i] & 0xffff);
238  if (op == spv::Op::OpVariable) {
239  int name = s.opcodes_[i + 1];
240  auto sc = spv::StorageClass(s.opcodes_[i + 3]);
241  Variable b;
242  b.debugName = debugNames[name];
243  b.name = name;
244  b.location = locations[name];
245  b.set = sets[name];
246  b.instruction = i;
247  b.storageClass = sc;
248  result.push_back(b);
249  }
250  }
251  return std::move(result);
252  }

521  // Create the buffer object without memory.

531 
532  // Create a memory object to bind to the buffer.

535  mai.memoryTypeIndex = vku::findMemoryTypeIndex(memprops, memreq.memoryTypeBits, memflags);
536  s.mem = device.allocateMemoryUnique(mai);

808  // Clear the colour of an image.
809  void clear(vk::CommandBuffer cb, const std::array<float,4> colour = {1, 1, 1, 1}) {
810  setLayout(cb, vk::ImageLayout::eTransferDstOptimal);
811  vk::ClearColorValue ccv(colour);
812  vk::ImageSubresourceRange range{vk::ImageAspectFlagBits::eColor, 0, 1, 0, 1};
813  cb.clearColorImage(*s.image, vk::ImageLayout::eTransferDstOptimal, ccv, range);
814  }
815 
816  // Update the image with an array of pixels. (Currently 2D only)
817  void update(vk::Device device, const void *data, vk::DeviceSize bytesPerPixel) {
818  vk::ImageSubresource subresource{vk::ImageAspectFlagBits::eColor, 0, 0};
819  auto srlayout = device.getImageSubresourceLayout(*s.image, subresource);
820 
821  uint8_t *dest = (uint8_t *)device.mapMemory(*s.mem, 0, s.size, vk::MemoryMapFlags{}) + srlayout.offset;
822  const uint8_t *src = (const uint8_t *)data;
823 
824  size_t bytesPerLine = s.extent.width * bytesPerPixel;
825  for (int y = 0; y != s.extent.height; ++y) {
826  memcpy(dest, src, bytesPerLine);
827  src += bytesPerLine;
828  dest += srlayout.rowPitch;
829  }
830  device.unmapMemory(*s.mem);
831  }
832 
833  // Copy another image to this one. This also changes the layout.
834  void copy(vk::CommandBuffer cb, vku::GenericImage &srcImage) {
835  srcImage.setLayout(cb, vk::ImageLayout::eTransferSrcOptimal);
836  setLayout(cb, vk::ImageLayout::eTransferDstOptimal);
837  vk::ImageCopy region{};
838  region.srcSubresource = {vk::ImageAspectFlagBits::eColor, 0, 0, 1};
839  region.dstSubresource = {vk::ImageAspectFlagBits::eColor, 0, 0, 1};
840  region.extent = s.extent;
841  cb.copyImage(srcImage.image(), vk::ImageLayout::eTransferSrcOptimal, *s.image, vk::ImageLayout::eTransferDstOptimal, region);
842  }
843 
844  // Change the layout of this image using a memory barrier.
845  void setLayout(vk::CommandBuffer cb, vk::ImageLayout newLayout, vk::ImageAspectFlags aspectMask = vk::ImageAspectFlagBits::eColor) {
846  vk::ImageLayout oldLayout = s.currentLayout;
847  s.currentLayout = newLayout;
848 
849  vk::ImageMemoryBarrier imageMemoryBarriers = {};
850  imageMemoryBarriers.srcQueueFamilyIndex = VK_QUEUE_FAMILY_IGNORED;
851  imageMemoryBarriers.dstQueueFamilyIndex = VK_QUEUE_FAMILY_IGNORED;
852  imageMemoryBarriers.oldLayout = oldLayout;
853  imageMemoryBarriers.newLayout = newLayout;
854  imageMemoryBarriers.image = *s.image;
855  imageMemoryBarriers.subresourceRange = {aspectMask, 0, 1, 0, 1};
856 
857  // Put barrier on top
858  vk::PipelineStageFlags srcStageMask{vk::PipelineStageFlagBits::eTopOfPipe};
859  vk::PipelineStageFlags dstStageMask{vk::PipelineStageFlagBits::eTopOfPipe};
860  vk::DependencyFlags dependencyFlags{};
861  vk::AccessFlags srcMask{};
862  vk::AccessFlags dstMask{};
863 
864  typedef vk::ImageLayout il;
865  typedef vk::AccessFlagBits afb;
866 
867  // Is it me, or are these the same?
868  switch (oldLayout) {
869  case il::eUndefined: break;
870  case il::eGeneral: srcMask = afb::eTransferWrite; break;
871  case il::eColorAttachmentOptimal: srcMask = afb::eColorAttachmentWrite; break;
872  case il::eDepthStencilAttachmentOptimal: srcMask = afb::eDepthStencilAttachmentWrite; break;
873  case il::eDepthStencilReadOnlyOptimal: srcMask = afb::eDepthStencilAttachmentRead; break;
874  case il::eShaderReadOnlyOptimal: srcMask = afb::eShaderRead; break;
875  case il::eTransferSrcOptimal: srcMask = afb::eTransferRead; break;
876  case il::eTransferDstOptimal: srcMask = afb::eTransferWrite; break;
877  case il::ePreinitialized: srcMask = afb::eTransferWrite|afb::eHostWrite; break;
878  case il::ePresentSrcKHR: srcMask = afb::eMemoryRead; break;
879  }
880 
881  switch (newLayout) {
882  case il::eUndefined: break;
883  case il::eGeneral: dstMask = afb::eTransferWrite; break;
884  case il::eColorAttachmentOptimal: dstMask = afb::eColorAttachmentWrite; break;
885  case il::eDepthStencilAttachmentOptimal: dstMask = afb::eDepthStencilAttachmentWrite; break;
886  case il::eDepthStencilReadOnlyOptimal: dstMask = afb::eDepthStencilAttachmentRead; break;
887  case il::eShaderReadOnlyOptimal: dstMask = afb::eShaderRead; break;
888  case il::eTransferSrcOptimal: dstMask = afb::eTransferRead; break;
889  case il::eTransferDstOptimal: dstMask = afb::eTransferWrite; break;
890  case il::ePreinitialized: dstMask = afb::eTransferWrite; break;
891  case il::ePresentSrcKHR: dstMask = afb::eMemoryRead; break;
892  }
893 
894  imageMemoryBarriers.srcAccessMask = srcMask;
895  imageMemoryBarriers.dstAccessMask = dstMask;
896  auto memoryBarriers = nullptr;
897  auto bufferMemoryBarriers = nullptr;
898  cb.pipelineBarrier(srcStageMask, dstStageMask, dependencyFlags, memoryBarriers, bufferMemoryBarriers, imageMemoryBarriers);
899  }
900 
901  vk::Format format() const { return s.format; }
902 protected:
903  void create(vk::Device device, const vk::PhysicalDeviceMemoryProperties &memprops, const vk::ImageCreateInfo &info, vk::ImageViewType viewType, vk::ImageAspectFlags aspectMask, bool hostImage) {
904  s.currentLayout = info.initialLayout;
905  s.extent = info.extent;
906  s.image = device.createImageUnique(info);
907  s.format = info.format;
908 
909  // Find out how much memory and which heap to allocate from.
910  auto memreq = device.getImageMemoryRequirements(*s.image);
911  vk::MemoryPropertyFlags search{};
912  if (hostImage) search = vk::MemoryPropertyFlagBits::eHostCoherent | vk::MemoryPropertyFlagBits::eHostVisible;
913 
914  // Create a memory object to bind to the buffer.
915  // Note: we don't expect to be able to map the buffer.
916  vk::MemoryAllocateInfo mai{};
917  mai.allocationSize = s.size = memreq.size;
918  mai.memoryTypeIndex = vku::findMemoryTypeIndex(memprops, memreq.memoryTypeBits, search);
919  s.mem = device.allocateMemoryUnique(mai);
920 
921  device.bindImageMemory(*s.image, *s.mem, 0);
922 
923  if (!hostImage) {
924  vk::ImageViewCreateInfo viewInfo{};
925  viewInfo.flags;
926  viewInfo.image = *s.image;
927  viewInfo.viewType = viewType;
928  viewInfo.format = info.format;
929  viewInfo.components = { vk::ComponentSwizzle::eR, vk::ComponentSwizzle::eG, vk::ComponentSwizzle::eB, vk::ComponentSwizzle::eA };
930  viewInfo.subresourceRange = vk::ImageSubresourceRange{aspectMask, 0, 1, 0, 1};
931  s.imageView = device.createImageViewUnique(viewInfo);
932  }
933  }
